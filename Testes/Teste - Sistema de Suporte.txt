create database suporte_tecnico;
use suporte_tecnico;

create table clientes(
id int auto_increment primary key,
nome varchar(50) not null
);

insert into clientes(nome) values ('João'), ('Marcos'), ('Pedro'), ('Aline');
			
create table chamados(
numero_chamado int auto_increment primary key,
data_chamdo datetime default current_timestamp,
descricao varchar(500) not null,
id_usuario int,
foreign key (id_usuario) references clientes(id)
);

create table atendentes(
id_atendent int auto_increment primary key,
nome_atendente varchar(45),
data_atendimento date not null,
descricao_servico varchar(500),
chamado_numero int,
foreign key (chamado_numero) references chamados(numero_chamado)
);
